$total_memory = $matches[1];
$used_memory = $matches[2];
$memory_percent = round(($used_memory / $total_memory) * 100);

// Check disk usage
$disk_usage = disk_free_space('/') / disk_total_space('/') * 100;
$disk_free_percent = round(100 - $disk_usage);

// Alert thresholds
$cpu_threshold = 80; // CPU load 
$memory_threshold = 85; // Memory usage percentage
$disk_threshold = 90; // Disk usage percentage

// Check if any threshold is exceeded
$alert_triggered = false;
$alert_message = "Server Resource Alert\n\n";
$alert_message .= "Site: $site_name\n";
$alert_message .= "Time: " . date('Y-m-d H:i:s') . "\n\n";

if ($cpu_usage > $cpu_threshold) {
    $alert_triggered = true;
    $alert_message .= "CPU ALERT: Load average is $cpu_usage (threshold: $cpu_threshold)\n";
}

if ($memory_percent > $memory_threshold) {
    $alert_triggered = true;
    $alert_message .= "MEMORY ALERT: Memory usage is $memory_percent% (threshold: $memory_threshold%)\n";
    $alert_message .= "Used: $used_memory MB / Total: $total_memory MB\n";
}

if ($disk_free_percent > $disk_threshold) {
    $alert_triggered = true;
    $alert_message .= "DISK ALERT: Disk usage is $disk_free_percent% (threshold: $disk_threshold%)\n";
}

// Send alert if needed
if ($alert_triggered) {
    $subject = "ALERT: $site_name - Server Resource Warning";
